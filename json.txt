            function testUser(){
                var XHR;

                if (window.XMLHttpRequest){
                    //For Mozilla and Non Microsoft Browsers
                    XHR=new XMLHttpRequest();
                }
                    else{
                    alert("XMLHttpRquest object not supported in your browser");
                    }

                XHR.open("GET", "data.json" ,true);
                XHR.onreadystatechange = function(){
                                displayMessage(XHR);
                             };
                XHR.send();
            }

            function displayMessage(XHR){
                if(XHR.readyState==4){
                    if(XHR.status==200){
                        var d=JSON.parse(XHR.responseText);

                        document.getElementById("result").innerHTML="<td>"+d[0].name+"</td>"+"<td>"+d[0].genre+"</td>"+"<td>"+d[0].credits.director+"</td>"+"<td>"+d[0].credits.year_released+"</td>";
                                        document.getElementById("result1").innerHTML="<td>"+d[1].name+"</td>"+"<td>"+d[1].genre+"</td>"+"<td>"+d[1].credits.director+"</td>"+"<td>"+d[1].credits.year_released+"</td>";
                    }
                    else{
                        document.getElementId("error").innerHTML=XHR.statusText;
                        document.getElementById("error").style.display="block";
                    }
                }
            }
        </script>
